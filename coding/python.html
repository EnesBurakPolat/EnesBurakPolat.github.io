# Basit bir sınıf örneği
class Kisi:
    def __init__(self, ad, yas):
        self.ad = ad
        self.yas = yas

    def tanit(self):
        return f"Adım {self.ad} ve yaşım {self.yas}"

kisi = Kisi("Ali", 30)
print(kisi.tanit())
